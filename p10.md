pipeline {
    agent any

    stages {

        stage('Checkout') {
            steps {
                git branch: 'master',
                    url: 'https://github.com/divya-1706/Simple-Calculator'
            }
        }

        stage('Verify') {
            steps {
                sh 'python3 --version'
                sh 'ls -la'
            }
        }

        stage('Run') {
            steps {
                sh 'python3 "Simple Calculator by Python.py" || true'
            }
        }

    }
}
